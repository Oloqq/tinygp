program = block

# kolejnosc nie ma znaczenia, priorytet jest jasny bo nie ma infixow
expr
    = ADD expr expr
    | SUB expr expr
    | MUL expr expr
    | DIV expr expr
    | LT expr expr
    | GT expr expr
    | OR expr expr
    | AND expr expr
    | NOT expr
    | reg

stat
    = LOAD reg expr
    | if
    | WHILE expr DO block END
    | INPUT reg
    | OUTPUT expr

block = stat*

if
    = IF expr block ELSE block END
    | IF expr block END

# reg to register/rejestr/zmienna
reg = R NUM
