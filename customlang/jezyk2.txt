program = block

# kolejnosc nie ma znaczenia, priorytet jest jasny bo nie ma infixow
expr
    = ADD expr expr
    | SUB expr expr
    | MUL expr expr
    | DIV expr expr
    | LT expr expr
    | GT expr expr
    | OR expr expr
    | AND expr expr
    | NOT expr
    | reg

stat
    = LOAD reg expr
    | IF expr block
    | WHILE expr DO block
    | INPUT reg
    | OUTPUT expr

block = stat* (ELSE | END)?

# reg to register/rejestr/zmienna
reg = R NUM
